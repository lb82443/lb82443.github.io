## Balls falling through pins will show you the Normal distribution.
library(animation)
ani.options(nmax = ifelse(interactive(), 100 + 14 - 2, 2), 
    interval = 0.03)
quincunx(balls = 100, layers = 14, col.balls = rainbow(200))
